async function askAI() { const input = document.getElementById('userInput'); const win = document.getElementById('chat-window'); const text = input.value.trim(); if(!text) return; win.innerHTML += `
${text}
`; input.value = ""; win.scrollTop = win.scrollHeight; try { // تعليمات جديدة لجوجل عشان يقسم الرد لجزئين const prompt = `أجب كمحامي مصري. قسم ردك لجزئين مفصولين بعلامة (###). الجزء الأول: نبذة مجانية مختصرة جداً عن الحالة (سطر واحد). الجزء الثاني: الرد القانوني الكامل والمفصل بالمواد القانونية. السؤال هو: ${text}`; const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${AI_KEY}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) }); const data = await res.json(); const fullReply = data.candidates[0].content.parts[0].text; // تقسيم الرد عن طريق العلامة ### const parts = fullReply.split('###'); const preview = parts[0] || "تم تحليل طلبك بنجاح."; const hiddenSecret = parts[1] || "التفاصيل القانونية جاهزة للعرض."; win.innerHTML += `
📋 رأي مبدئي:
${preview}
${hiddenSecret}
🔒 الرد القانوني المفصل محمي إتمام الدفع يفك التشفير فوراً
`; document.getElementById('payment-box').style.display = "block"; win.scrollTop = win.scrollHeight; } catch (e) { win.innerHTML += `
🔒 الرد القانوني جاهز، يرجى الدفع لعرضه.
`; document.getElementById('payment-box').style.display = "block"; } }
